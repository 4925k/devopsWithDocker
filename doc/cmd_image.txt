#### IMAGE  ###

# list images
$ docker image ls 

# pull image
$ docker image pull <image>

# remove image
$ docker image rm <image>

# prune images
$ docker image prune

# pull image with given tag
$ docker pull <image>:<tag>

# tag image locally 
$ docker tag <image>:<tag> <new_image>:<new_tag>

# save changes done to a container as an image
$ docker commit <container_name> <new_image_name>
